ApplyCardEffect의 switch(effectType)가 데미지·방어·다단히트·관통·회복 등 18종 효과를 분기 — 새 카드는 데이터 한 장 추가로 끝납니다.
인스턴스 단위 손패 관리
같은 CardData가 여러 장이어도 handCardIds로 구분해, 시선 비용 보정을 정확한 카드에만 적용합니다.
데미지 합성 파이프라인
CalculateDamage가 근력 가산 → 시선 배율 → 약화 25% 감소를 순서대로 합성합니다.
BattleManager.cs
// 카드 사용 — 비용 확인 → 시선 누적 → 효과 적용
public bool PlayCardOnMonster(CardData card, Monster target, int id = 0) {
    int handIndex = FindHandIndex(card, id);
    if (handIndex < 0 || !target.IsAlive) return false;

    int cost = GetCardCost(card, id);
    if (currentMana < cost) return false;   // 마나 부족

    currentMana -= cost;
    RemoveFromHandAt(handIndex);
    GazeEffectManager.Instance?.OnCardPlayed(card, true);

    discardPile.Add(card);
    ApplyCardEffect(card, target);  // 데이터 주도 분기
    return true;
}

// 새 상태 = enum 항목 + case 1개
public static Data Get(StatusType t) {

    // Weak / Regen / OpenEye … 같은 패턴으로 추가

// anchor 기준 좌우 균등 분산 — 일반 여러 마리 배치
foreach (var entry in encounter.entries) {
    if (entry?.data == null) continue;
    // 단독=0, 2마리=±s/2, 3마리=-s/0/+s
    float autoX = (idx++ - (validCount-1)*0.5f) * monsterSpacing;
    Vector3 pos = monsterAnchorPosition + new Vector3(autoX, 0, 0);
    var go = Instantiate(monsterPrefab, pos, Quaternion.identity);
    go.GetComponent<Monster>().data = entry.data;
    AttachBossAI(go, entry.data);   // 엘리트=전용 AI
}
